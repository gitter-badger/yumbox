app = angular.module 'app', ['ngRoute', 'ngFlowGrid'] app.controller 'AppC', [ '$scope', 'fgDelegate', ($scope, $mdSidenav, fgDelegate)-> $.material.init() ] app.config [ '$routeProvider', '$locationProvider', ($routeProvider, $locationProvider) -> $locationProvider.hashPrefix '!' $routeProvider. when('/', { templateUrl: 'views/components/home/index' controller: 'homeCtrl' }).when('/:page', { templateUrl: ($routeParams) -> "views/components/#{$routeParams.page}/index" controller: [ '$scope', '$routeParams', '$controller', ($scope, $routeParams, $controller) -> $controller("#{$routeParams.page}Ctrl", {$scope:$scope}) ] }) ]
