Path = require 'path' exports.register = (plugin, options, next) -> server = plugin.select 'web' server.views { engines: { jade: require('jade') }, path: Path.join(__dirname, 'app') } server.register { register: require('hapi-assets'), options: require('./assets') }, (err) -> console.log('Failed loading plugin hapi-assets in web plugin') if(err) server.route require './routes' next() exports.register.attributes = { pkg: require('../package.json') }
